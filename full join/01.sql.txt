CREATE TABLE ALUNO (CODALUNO INT, NOMEALUNO VARCHAR(50), CODCURSO INT, PRIMARY KEY(CODALUNO), FOREIGN KEY(CODCURSO) REFERENCES CURSO(CODCURSO));
